panel.cor = function(x, y, digits=2, prefix="", cex.cor){
    usr = par("usr"); on.exit(par(usr))
    par(usr = c(0, 1, 0, 1))
    r = abs(cor(x, y, use = "complete.obs"))   # use complete if no data values
    txt = format(c(r, 0.123456789), digits=digits)[1]
    txt = paste(prefix, "R = ", txt, sep="")
    if(missing(cex.cor)) cex.cor = 0.8/strwidth(txt)
    text(0.5, 0.65, txt, cex = cex.cor * r)
    txt = format(c(E, 0.123456789), digits=digits)[1]
    txt = paste(prefix, "E = ", txt, sep="")
    text(0.5, 0.35, txt, cex = cex.cor * E)	
}
